\begin{document}
	\pgfmathsetmacro\layersize{120}%
	\pgfmathsetmacro\layerinitialrotation{0}%

	\begin{tikzpicture}[x=1pt,y=1pt,>=Latex]
		\tikzset{triangleface/.style={
			fill=yellow!35
		}}
		\tikzset{squareface/.style={
			fill=orange!35
		}}
		\tikzset{line/.style={
			thick
		}}
		\tikzset{layersline/.style={
			line,
			black
		}}
		\tikzset{nearestline/.style={
			line,
			Green
		}}
		\tikzset{squaresplitline/.style={
			line,
			Red
		}}

		\coordinate (center) at (0,0);

		%---------------------------------
		% Angle
		\uncover<2-3>{
			\filldraw[DarkGreen!70] (center) -- (30,0) arc (0:60:30) -- cycle;
		}
		\uncover<2>{
	    	\node at (20:50)  {$\theta = \frac{i\pi}{n}$};
		}
		\uncover<3-3>{
	    	\node at (20:50)  {$\theta = \frac{i\pi}{6}$};
		}

	    %---------------------------------
		% Points
		\foreach \i in {0,...,5} {
			\coordinate (P\i) at (\layerinitialrotation+\i*360/6:\layersize);
		};

	    %---------------------------------
		% Lines
		\uncover<2-4>{
			\foreach \i in {0,...,5} {
				\draw[layersline] (center) -- (P\i);
			};
		}

		%---------------------------------
		% Numbers
		\uncover<3->{
			\node[right] at (P0) {0};
			\node[above right] at (P1) {1};
			\node[above left] at (P2) {2};
			\node[left] at (P3) {3};
			\node[below left] at (P4) {4};
			\node[below right] at (P5) {5};
		}

		%---------------------------------
		% Face lines
		\uncover<1-4>{
			\draw[layersline] (0,0) circle (120);
		}
		\uncover<4-5>{
			\draw[layersline] (P0) -- (P1) -- (P2) -- (P3) -- (P4) -- (P5) -- cycle;
		}
	\end{tikzpicture}
\end{document}
